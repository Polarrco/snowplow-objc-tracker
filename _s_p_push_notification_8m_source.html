182 - (NSDictionary *) payload {
183  NSMutableDictionary * event = [[NSMutableDictionary alloc] init];
184  [event setObject:_title forKey:kSPPnTitle];
185  [event setObject:_body forKey:kSPPnBody];
186  [event setValue:_badge forKey:kSPPnBadge];
187  if (_subtitle != nil) {
188  [event setObject:_subtitle forKey:kSPPnSubtitle];
189  }
190  if (_subtitle != nil) {
191  [event setObject:_subtitle forKey:kSPPnSubtitle];
192  }
193  if (_sound != nil) {
194  [event setObject:_sound forKey:kSPPnSound];
195  }
196  if (_launchImageName != nil) {
197  [event setObject:_launchImageName forKey:kSPPnLaunchImageName];
198  }
199  if (_userInfo != nil) {
200  NSMutableDictionary * aps = nil;
201  NSMutableDictionary * newUserInfo = nil;
202 
203  // modify contentAvailable value "1" and "0" to @YES and @NO to comply with schema
204  if (![[_userInfo valueForKeyPath:@"aps.contentAvailable"] isEqual:nil] &&
205  [[_userInfo objectForKey:@"aps"] isKindOfClass:[NSDictionary class]]) {
206  aps = [[NSMutableDictionary alloc] initWithDictionary:_userInfo[@"aps"]];
207 
208  if ([[_userInfo valueForKeyPath:@"aps.contentAvailable"] isEqual:@1]) {
209  [aps setObject:@YES forKey:@"contentAvailable"];
210  } else if ([[_userInfo valueForKeyPath:@"aps.contentAvailable"] isEqual:@0]) {
211  [aps setObject:@NO forKey:@"contentAvailable"];
212  }
213  newUserInfo = [[NSMutableDictionary alloc] initWithDictionary:_userInfo];
214  [newUserInfo setObject:aps forKey:@"aps"];
215  }
216  [event setObject:[[NSDictionary alloc] initWithDictionary:newUserInfo] forKey:kSPPnUserInfo];
217  }
218  if (_attachments != nil) {
219  [event setObject:_attachments forKey:kSPPnAttachments];
220  }
221 
222  return [[NSDictionary alloc] initWithDictionary:event copyItems:YES];
223 }
