167 - (SPEmitterEvent *) getEventWithId:(long long int)id_ {
168  __block SPEmitterEvent *event = nil;
169  [self.queue inDatabase:^(FMDatabase *db) {
170  if ([db open]) {
171  FMResultSet *s = [db executeQuery:_querySelectId, [NSNumber numberWithLongLong:id_]];

173  NSData * data = [s dataForColumn:@"eventData"];
