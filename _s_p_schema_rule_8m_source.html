59 - (BOOL)matchWithUri:(NSString *)uri {
60  if (!uri) {
61  return NO;
62  }
63  NSArray<NSString *> *uriParts = [self partsFromUri:uri regexPattern:kUriPattern];
64  if (uriParts.count < _ruleParts.count) {
65  return NO;
66  }
67  // Check vendor part
68  NSArray<NSString *> *ruleVendor = [_ruleParts[0] componentsSeparatedByString:@"."];
69  NSArray<NSString *> *uriVendor = [uriParts[0] componentsSeparatedByString:@"."];
70  if (uriVendor.count != ruleVendor.count) {
71  return NO;
72  }
73  NSUInteger index = 0;
74  for (NSString *ruleVendorPart in ruleVendor) {
75  if (![@"*" isEqualToString:ruleVendorPart] && ![uriVendor[index] isEqualToString:ruleVendorPart]) {
76  return NO;
77  }
78  index++;
79  }
80  // Check the rest of the rule
81  index = 1;
82  for (NSString *rulePart in [_ruleParts subarrayWithRange:NSMakeRange(1, _ruleParts.count-1)]) {
83  if (![@"*" isEqualToString:rulePart] && ![uriParts[index] isEqualToString:rulePart]) {
84  return NO;
85  }
86  index++;
87  }
88  return YES;
89 }

93 - (nullable NSArray<NSString *> *)partsFromUri:(NSString *)uri regexPattern:(NSString *)pattern {
94  NSError *error = NULL;
95  NSRegularExpression *regex = [NSRegularExpression regularExpressionWithPattern:pattern options:0 error:&error];
96  NSTextCheckingResult *match = [regex firstMatchInString:uri options:0 range:NSMakeRange(0, uri.length)];
97  NSMutableArray *parts = [NSMutableArray arrayWithCapacity:6];
98  if (!match) {
99  return nil;
100  }
101  for (int i = 1; i < match.numberOfRanges; i++) {
102  if (i > 6) {
103  return nil;
104  }
105  NSString *part = [uri substringWithRange:[match rangeAtIndex:i]];
106  [parts setObject:part.copy atIndexedSubscript:i-1];
107  }
108  return parts;
109 }

111 - (BOOL)validateVendor:(NSString *)vendor {
112  // the components array will be generated like this from vendor:
113  // "com.acme.marketing" => ["com", "acme", "marketing"]
114  NSArray<NSString *> *components = [vendor componentsSeparatedByString:@"."];
115  // check that vendor doesn't begin or end with period
116  // e.g. ".snowplowanalytics.snowplow." => ["", "snowplowanalytics", "snowplow", ""]
117  if (components.count > 1 && (!components[0].length || !components[components.count-1].length)) {
118  return NO;
119  }
120  // reject vendors with criteria that are too broad & don't make sense, i.e. "*.*.marketing"
121  if ([@"*" isEqualToString:components[0]] || [@"*" isEqualToString:components[1]]) {
122  return NO;
123  }
124  // now validate the remaining parts, vendors should follow matching that never breaks trailing specificity
125  // in other words, once we use an asterisk, we must continue using asterisks for parts or stop
126  // e.g. "com.acme.marketing.*.*" is allowed, but "com.acme.*.marketing.*" or "com.acme.*.marketing" is forbidden
127  if (components.count <= 2) return YES;
128  // trailingComponents are the remaining parts after the first two
129  NSArray<NSString *> *trailingComponents = [components subarrayWithRange:NSMakeRange(2, components.count-2)];
130  BOOL asterisk = NO;
131  for (NSString *part in trailingComponents) {
132  if ([@"*" isEqualToString:part]) { // mark when we've found a wildcard
133  asterisk = true;
134  } else if (asterisk) { // invalid when alpha parts come after wildcard
135  return NO;
136  }
137  }
138  return YES;
139 }
