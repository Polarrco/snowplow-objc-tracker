91 - (void)log:(SPLogLevel)level tag:(NSString *)tag message:(NSString *)message {
92  if (level > self.logLevel) {
93  return;
94  }
95  if (self.delegate) {
96  switch (level) {
97  case SPLogLevelOff:
98  // do nothing.
99  break;
100  case SPLogLevelError:
101  [self.delegate error:tag message:message];
102  break;
103  case SPLogLevelDebug:
104  [self.delegate debug:tag message:message];
105  break;
106  case SPLogLevelVerbose:
107  [self.delegate verbose:tag message:message];
108  break;
109  }
110  return;
111  }
112  #if SNOWPLOW_TEST
113  // NSLog doesn't work on test target
114  NSString *output = [NSString stringWithFormat:@"[%@] %@: %@", @[@"Off", @"Error", @"Error", @"Debug", @"Verbose"][level], tag, message];
115  printf("%s", [output UTF8String]);
116  #elif DEBUG
117  // Log should be printed only during debugging
118  NSLog(@"[%@] %@: %@", @[@"Off", @"Error", @"Debug", @"Verbose"][level], tag, message);
119  #endif
120 }

122 - (void)trackErrorWithTag:(NSString *)tag message:(NSString *)message errorOrException:(id)errorOrException {
123  if (self.errorLogger) {
