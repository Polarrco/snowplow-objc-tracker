128 - (SPScreenType) _SP_getViewControllerType:(UIViewController *)viewController {
129  if ([viewController isKindOfClass:[UINavigationController class]]) {
130  return SPScreenTypeNavigation;
131  } else if ([viewController isKindOfClass:[UITabBarController class]]) {
132  return SPScreenTypeTabBar;
133  } else if (viewController.presentedViewController) {
134  return SPScreenTypeModal;
135  } else if ([viewController isKindOfClass:[UIPageViewController class]]) {
136  return SPScreenTypePageView;
137  } else if ([viewController isKindOfClass:[UIPopoverPresentationController class]]) {
138  return SPScreenTypePopoverPresentation;
139  } else if ([viewController isKindOfClass:[UISplitViewController class]]) {
140  return SPScreenTypeSplitView;
141  } else {
142  return SPScreenTypeDefault;

150 - (UIViewController *) _SP_topViewController {
151  UIWindow *keyWindow = nil;
152  NSArray<UIWindow *> *windows = [UIApplication sharedApplication].windows;
153  for (UIWindow *window in windows) {
154  if (window.isKeyWindow) {
155  keyWindow = window;
156  break;
157  }
158  }
159  if (!keyWindow) {
160  return nil;
161  }
162  return [self _SP_topViewController:keyWindow.rootViewController];
163 }

165 - (UIViewController *) _SP_topViewController:(UIViewController *)rootViewController {
166  if ([rootViewController isKindOfClass:[UINavigationController class]]) {
167  UINavigationController *navigationController = (UINavigationController *)rootViewController;
168  return [self _SP_topViewController:[navigationController.viewControllers lastObject]];
169  }
170 
171  if ([rootViewController isKindOfClass:[UITabBarController class]]) {
172  UITabBarController *tabController = (UITabBarController *)rootViewController;
173  return [self _SP_topViewController:tabController.selectedViewController];
174  }
175 
176  UIViewController *presentedViewController = rootViewController.presentedViewController;
177  if (presentedViewController != nil) {
178  return [self _SP_topViewController:presentedViewController];
179  }
180 
181  return rootViewController;
