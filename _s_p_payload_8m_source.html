70 - (void) addJsonToPayload:(NSData *)json
71  base64Encoded:(Boolean)encode
72  typeWhenEncoded:(NSString *)typeEncoded
73  typeWhenNotEncoded:(NSString *)typeNotEncoded {
74  NSError *error = nil;
75  NSDictionary *object = nil;
76  @try {
77  object = [NSJSONSerialization JSONObjectWithData:json options:0 error:&error];
78  }
79  @catch (NSException *exception) {
80  SPLogPayloadError(exception, @"Json to payload exception, %@", exception.name);
81  return;
82  }
83  if (error) {
84  SPLogPayloadError(error, @"Json to payload error, %@", error);
85  return;
86  }
87  if (![object isKindOfClass:[NSDictionary class]]) {
88  SPLogPayloadError(nil, @"Serialized json isn't a NSDictionary type");
89  return;
90  }
91  if (encode) {
92  NSString *encodedString = [json base64EncodedStringWithOptions:0];
93 
94  // We need URL safe with no padding. Since there is no built-in way to do this, we transform
95  // the encoded payload to make it URL safe by replacing chars that are different in the URL-safe
96  // alphabet. Namely, 62 is - instead of +, and 63 _ instead of /.
97  // See: https://tools.ietf.org/html/rfc4648#section-5
98  encodedString = [[encodedString stringByReplacingOccurrencesOfString:@"/" withString:@"_"]
99  stringByReplacingOccurrencesOfString:@"+" withString:@"-"];
100 
101  // There is also no padding since the length is implicitly known.
102  encodedString = [encodedString stringByTrimmingCharactersInSet:[NSCharacterSet characterSetWithCharactersInString:@"="]];
103 
104  [self addValueToPayload:encodedString forKey:typeEncoded];
105  } else {
106  [self addValueToPayload:[[NSString alloc] initWithData:json encoding:NSUTF8StringEncoding] forKey:typeNotEncoded];
107  }
108 }
