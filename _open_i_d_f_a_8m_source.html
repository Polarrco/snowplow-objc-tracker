55 + (NSArray*) threeDaysOpenIDFAArray {
56 
57  // The following list represents a rather large array of ids used to detect
58  // the presence of a number of apps available on the device. The apps on this
59  // list have been handpicked by Appsfire AppGenome engine; they each carry
60  // a statistically significant weight to define and differentiate a user from
61  // another. For the purpose of obfuscation inside the lib, we only used apps
62  // with Facebook-related URLHanders typically starting with the string "fb"
63  // then followed by a numerical string.
64  //
65  // RATIONALE: an "appmap" profile may vary over time, however, the likeness that it
66  // evolves between two tracking events over a large proportion of the user
67  // base is low.
68  //
69  // NOTICE: this list is the property of Appsfire and may not be extracted from
70  // the context of OpenIDFA which is governed by a Creative Commons (No Derivatives)
71  //
72  NSArray* base = @[ @101015295179ll, @102443183283204ll, @105130332854716ll, @110633906966ll, @111774748922919ll, @112953085413703ll, @113174082133029ll, @113246946530ll, @114870218560647ll, @115829135094686ll, @115862191798713ll, @118506164848956ll, @118589468194837ll, @118881298142408ll, @120176898077068ll, @121848807893603ll, @123448314320ll, @123591657714831ll, @124024574287414ll, @127449357267488ll, @127995567256931ll, @132363533491609ll, @134841016914ll, @138326442889677ll, @138713932872514ll, @146348772076164ll, @147364571964693ll, @147712241956950ll, @148327618516767ll, @152777738124418ll, @154615291248069ll, @156017694504926ll, @158761204309396ll, @159248674166087ll, @160888540611569ll, @161500167252219ll, @161599933913761ll, @162729813767876ll, @165482230209033ll, @176151905794941ll, @177821765590225ll, @178508429994ll, @192454074134796ll, @194714260574159ll, @208559595824260ll, @209864595695358ll, @210068525731476ll, @239823722730183ll, @246290217488ll, @255472420580ll, @267160383314960ll, @292065597989ll, @99197768694ll, @342234513642ll, @349724985922ll, @99554596360ll, @40343401983ll, @500407959994978ll, @52216716219ll, @90376669494ll];

75  // Playing with the prefix some more for the purpose of obfuscation
76  //
77  NSMutableString* _s_appmap = [NSMutableString stringWithString:@""];
78  NSString* _s = @"/";
79  NSString* _c = @":";
80  NSString* _b = @"b";
81  NSString* _f = @"f";
82 
83  // Computing the "appmap" profile string for this user
84  // STRENGTH: potential of 2^61 distinct combinations
85  //
86  [_s_appmap appendString:[self.class canOpenURL:[NSURL URLWithString:[NSString stringWithFormat:@"%@%@%@%@%@",_f,_b,_c,_s,_s]]]?@"|":@"-"];
87  for (id baseid in base) {
88  [_s_appmap appendString:[self.class canOpenURL:[NSURL URLWithString:[NSString stringWithFormat:@"%@%@%@%@%@%@",_f,_b,[baseid stringValue],_c,_s,_s]]]?@"|":@"-"];
89  }

91  // Collecting the device boottime (Unix epoch) and turn it into a string
92  //
93  // RATIONALE: any reboot will cause former OpenIDFAs for this user to be
94  // invalidated and useless. This is a built-in property of OpenIDFA that
95  // wants to ensure that tokens expire regularly, and in this case, non-
96  // determiniscally.
97  //
98  // COMPLEXITY: boottime is NOT stable. Few days after releasing the lib,
99  // I realized that the clock was adjusted by 2 seconds per day on an actual
100  // device probably due to real-time clock adjustments over NTP.
101  //
102  // SOLUTION: truncating the right hand side of the boottime by 4 digits
103  // This allows the boottime to slide by up to 9999 seconds... many days!
104  // Side benefits, reboots within a 2.77h window will NOT reset the IDFA
105  //
106  // STRENGTH: iOS devices do not reboot often (ahem, version 7.1+!)
107  // Typical reboot occurs days if not weeks apart. Using one week,
108  // gives us 600k distinct possibilities assuming a random distribution
109  // of reboots across all devices
110  //
111 
112  size_t size;
113  struct timeval boottime;
114 
115  int mib[2] = {CTL_KERN, KERN_BOOTTIME};
116  size = sizeof(boottime);
117  NSString* _s_bt = @"";
118  if (sysctl(mib, 2, &boottime, &size, NULL, 0) != -1 && boottime.tv_sec != 0) {
119  _s_bt =[NSString stringWithFormat:@"%lu",boottime.tv_sec];
120  _s_bt = [_s_bt substringToIndex:[_s_bt length]-4];
121  }
122 
123  // Collecting the device "machine" identifier
124  //
125  // STRENGTH: 20+ combinations
126  //
127  // WEAKNESS: compared to Android, iOS pales here :) oh wait...
128  //
129  sysctlbyname("hw.machine", NULL, &size, NULL, 0);
130  char *machine = malloc(size);
131  sysctlbyname("hw.machine", machine, &size, NULL, 0);
132  NSString *_s_machine = [NSString stringWithCString:machine encoding:NSUTF8StringEncoding];
133  free(machine);
134 
135  // Collecting the device "model" identifier, for completition
136  //
137  // STRENGTH: adds little, helps in case of simulator
138  //
139  sysctlbyname("hw.model", NULL, &size, NULL, 0);
140  char *model = malloc(size);
141  sysctlbyname("hw.model", model, &size, NULL, 0);
142  NSString *_s_model = [NSString stringWithCString:model encoding:NSUTF8StringEncoding];
143  free(model);
144 
145  // Collecting the locale country code
146  //
147  // STRENGTH: its complicated, but at least 80 combinations not evenly distributed!
148  // Check CLDR release 24 to know more: http://cldr.unicode.org/index/downloads/cldr-24
149  //
150  NSString *_s_ccode = [[NSLocale currentLocale] objectForKey: NSLocaleCountryCode];
151 
152  // Collecting an ordered array of preferred languages
153  //
154  // RATIONALE: this is an ordered list of preferred languages, theoretically speaking
155  // Most people only configure one and thefore the rest of the list remains unchanged
156  // However, polyglots will impact this list deeply.
157  // I've chosen to only mark up to 8 languages. Infiniglots don't pass the Turing test.
158  //
159  // STRENGTH: something like 8! but not evenly distributed.
160  //
161  NSArray* preferredLang = [NSLocale preferredLanguages];
162  NSString* _s_langs;
163  if (preferredLang == nil || ![preferredLang isKindOfClass:[NSArray class]] || [ preferredLang count ] == 0)
164  _s_langs = @"en";
165  else
166  _s_langs = [[preferredLang subarrayWithRange:NSMakeRange(0, MIN(8,[preferredLang count]))] componentsJoinedByString:@""];
167 
168  // Collecting a few more device specific discriminating strings
169  //
170  // RATIONALE: the goal is to reinfoce uniqueness / reduce collisions.
171  // 1/ memory size: iOS devices come in variations of 8Gb, 16Gb, 32Gb, and 64Gb
172  // 2/ OS version: unevenly distributed, most people tend to have the latest version
173  // in iOS world... which is good! distinctiveness value = close to zero
174  // 3/ timezone: assuming equal distribution, 24 variations. Though we intuitively
175  // all know which 8 to 10 timezones represent 90% of all iOS devices!
176  //
177  // STRENGTH: say 50 combinations.
178  //
179  NSDictionary *fattributes = [[NSFileManager defaultManager] attributesOfFileSystemForPath:NSHomeDirectory() error:nil];
180  NSString *_s_disk = [[fattributes objectForKey:NSFileSystemSize] stringValue];
181  NSString* _s_osv = [[UIDevice currentDevice] systemVersion];
182  NSString* _s_tmz = [[NSTimeZone systemTimeZone] name];
183 
184  // Marking the day, sameDay™, tomorrow or day after
185  //
186  // RATIONALE: this is another built-in and unique property of OpenIDFA
187  // The token is only valid for today (and the next two days).
188  // Tthe notion of day is pegged to the user himself, to their own
189  // biological clock. Assumption and experience is that most transactions
190  // occur within the same waking hours. So the notion of day was shifted
191  // by 4 hours (i.e. day begins at 4am, ends at 4am next day).
192  // The following study/graph by MixPanel illustrates this quite well:
193  // http://tctechcrunch2011.files.wordpress.com/2014/02/day-in-mobile-california.png?w=1280
194  // When tracking precision requires more than sameDay™, then tracker
195  // can venture into getting 3 days worth of OpenIDFA tokens...
196  //
197  NSDateFormatter* dateFormatter = [ [ NSDateFormatter alloc ] init ];
198  [ dateFormatter setDateFormat:@"yyMMdd" ];
199  NSCalendar *calendar;
200  calendar = [[NSCalendar alloc] initWithCalendarIdentifier:NSCalendarIdentifierGregorian];
201 
202  NSDateComponents *hourShift = [[NSDateComponents alloc] init];
203  [hourShift setHour:-4];
204  NSDate *currentDay= [calendar dateByAddingComponents:hourShift toDate:[NSDate date] options:0];
205  NSDateComponents *dayShift = [[NSDateComponents alloc] init];
206  [dayShift setDay:1];
207 
208  // Creating array of 3 OpenIDFAs
209  //
210  NSMutableArray* openIDFAs = [NSMutableArray arrayWithCapacity:3];
211  for (int j=0; j<3; j++) {
212 
213  // Creating the fingerprint with the various elements
214  // PART 1: Boot Time
215  // PART 2: Mostly stable and unique elements over time
216  // PART 3: Day Stamp
217  //
218  NSString* _s_day = [dateFormatter stringFromDate:currentDay];
219  NSString* fingerprint = [NSString stringWithFormat:@"%@ %@ %@ %@ %@ %@ %@ %@ %@ %@",
220  _s_bt, _s_disk,
221  _s_machine, _s_model,
222  _s_osv, _s_ccode,
223  _s_langs, _s_appmap,
224  _s_tmz, _s_day];
225 
226  const char* str = [fingerprint UTF8String];
227  unsigned char result[CC_SHA256_DIGEST_LENGTH];
228  CC_SHA256(str, (CC_LONG)strlen(str), result);
229 
230  // Creating a UUID-like formatting by only taking one byte out of two from
231  // the SHA-256 Hash, and by inserting dashes where relevant.
232  // Should anyone see benefits in better compliance with http://www.itu.int/rec/T-REC-X.667/en
233  // especially perhaps the version number represented by bits 49 to 52, then raise your hand!
234  //
235  NSMutableString *hash = [NSMutableString stringWithCapacity:36];
236  for(int i = 0; i<CC_SHA256_DIGEST_LENGTH; i+=2)
237  {
238  if (i==8 || i==12 || i==16 || i==20)
239  [hash appendString:@"-"];
240  [hash appendFormat:@"%02X",result[i]];
241  }
242 
243  [openIDFAs addObject:hash];
244 
245  //NSLog(@"day[%d]%@",j,fingerprint);
246  //NSLog(@"HASH%d: %@",j,hash);
247 
248  // Shifting day by one
249  //
250  currentDay = [calendar dateByAddingComponents:dayShift toDate:currentDay options:0];
251  }
252 
253  //NSLog(@"OpenIDFAs: %@",openIDFAs);
254  return openIDFAs;
255 }
