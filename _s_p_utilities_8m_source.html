101 /*
102  The IDFA can be retrieved using selectors rather than proper instance methods because
103  the compiler would complain about the missing AdSupport framework.
104  As stated in the header file, this only works if you have the AdSupport library in your project.
105  If you have it, but do not want to use IDFA, add the compiler flag <code>SNOWPLOW_NO_IFA</code> to your build settings.
106  If you haven't AdSupport framework in your project it just compiles returning a nil advertisingIdentifier.
107 
108  Note that `advertisingIdentifier` returns a sequence of 0s when used in the simulator.
109  Use a real device if you want a proper IDFA.
110  */
111 + (NSString *) getAppleIdfa {
112 #if SNOWPLOW_TARGET_IOS || SNOWPLOW_TARGET_TV
113 #ifndef SNOWPLOW_NO_IFA
114  Class ASIdentifierManagerClass = NSClassFromString(@"ASIdentifierManager");
115  if (!ASIdentifierManagerClass) return nil;
116 
117  SEL sharedManagerSelector = NSSelectorFromString(@"sharedManager");
118  if (![ASIdentifierManagerClass respondsToSelector:sharedManagerSelector]) return nil;
119 
120  id sharedManager = ((id (*)(id, SEL))[ASIdentifierManagerClass methodForSelector:sharedManagerSelector])(ASIdentifierManagerClass, sharedManagerSelector);
121 
122  SEL isAdvertisingTrackingEnabledSelector = NSSelectorFromString(@"isAdvertisingTrackingEnabled");
123  if (![sharedManager respondsToSelector:isAdvertisingTrackingEnabledSelector]) return nil;
124 
125  BOOL isAdvertisingTrackingEnabled = ((BOOL (*)(id, SEL))[sharedManager methodForSelector:isAdvertisingTrackingEnabledSelector])(sharedManager, isAdvertisingTrackingEnabledSelector);
126  if (!isAdvertisingTrackingEnabled) return nil;
127 
128  SEL advertisingIdentifierSelector = NSSelectorFromString(@"advertisingIdentifier");
129  if (![sharedManager respondsToSelector:advertisingIdentifierSelector]) return nil;
130 
131  NSUUID *uuid = ((NSUUID* (*)(id, SEL))[sharedManager methodForSelector:advertisingIdentifierSelector])(sharedManager, advertisingIdentifierSelector);
132  return [uuid UUIDString];
133 #endif
134 #endif
135  return nil;
136 }

413 + (NSString *) camelcaseParsedKey:(NSString *)key {
414  NSScanner * scanner = [[NSScanner alloc] initWithString:key];
415  NSMutableArray * words = [[NSMutableArray alloc] init];
416  NSString * scannedWord = [[NSString alloc] init];
417 
418  while (![scanner isAtEnd]) {
419  [scanner scanUpToString:@"-" intoString:&scannedWord];
420  [words addObject:scannedWord];
421  SPLogVerbose(@"scanned word: %@", scannedWord);
422  [scanner scanString:@"-" intoString:nil];
423  }
424 
425  SPLogVerbose(@"%@", words);
426  if ([words count] == 0) {
427  return @"";
428  } else if ([words count] == 1) {
429  return [[NSString alloc] initWithString:[words[0] lowercaseString]];
430  } else {
431  NSMutableString * camelcaseKey = [[NSMutableString alloc] initWithString:[words[0] lowercaseString]];
432  NSRange range;
433  range.length = words.count-1;
434  range.location = 1;
435  for (NSString * word in [words subarrayWithRange:range]) {
436  [camelcaseKey appendString:[word capitalizedString]];
437  }
438  return camelcaseKey;
439  }
440 }
