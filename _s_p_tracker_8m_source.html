73 void uncaughtExceptionHandler(NSException *exception) {
74  NSArray* symbols = [exception callStackSymbols];
75  NSString * stackTrace = [NSString stringWithFormat:@"Stacktrace:\n%@", symbols];
76  NSString * message = [exception reason];
77  dispatch_sync(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
78  // Load values
79  NSUserDefaults * userDefaults = [NSUserDefaults standardUserDefaults];
80  NSString * url = [userDefaults objectForKey:kSPErrorTrackerUrl];
81  NSString * protocolString = [userDefaults objectForKey:kSPErrorTrackerProtocol];
82  NSString * methodString = [userDefaults objectForKey:kSPErrorTrackerMethod];
83  SPProtocol protocol = SPHttps;
84  if (protocolString && [protocolString isEqual:@"http://"]) {
85  protocol = SPHttp;
86  }
87  SPRequestOptions method = SPRequestPost;
88  if (methodString && [methodString isEqual:@"/i"]) {
89  method = SPRequestGet;
90  }
91  // Send notification to tracker
92  SPEmitter * emitter = [SPEmitter build:^(id<SPEmitterBuilder> builder) {
93  [builder setUrlEndpoint:url];
94  [builder setProtocol:protocol];
95  [builder setHttpMethod:method];
96  }];
97  SPTracker * tracker = [SPTracker build:^(id<SPTrackerBuilder> builder) {
98  [builder setEmitter:emitter];
99  }];
100 
101  if (message == nil || [message length] == 0) {
102  return;
103  }
104  SNOWError * error = [SNOWError build:^(id<SPErrorBuilder> builder) {
105  [builder setMessage:message];
106  if (stackTrace != nil && [stackTrace length] > 0) {
107  [builder setStackTrace:stackTrace];
108  }
109  }];
110  [tracker track:error];
111  [NSThread sleepForTimeInterval:2.0f];
112  });
113 }

201 - (void) checkInstall {
202  SPInstallTracker * installTracker = [[SPInstallTracker alloc] init];
203  NSNumber * previousTimestamp = [installTracker getPreviousInstallTimestamp];
204  if (_installEvent) {
205  if (installTracker.isNewInstall) {
206  SPSelfDescribingJson * installEvent = [[SPSelfDescribingJson alloc] initWithSchema:kSPApplicationInstallSchema andData:@{}];
207  SPUnstructured * event = [SPUnstructured build:^(id<SPUnstructuredBuilder> builder) {
208  [builder setEventData:installEvent];
209  }];
210  [self track:event];
211  if (previousTimestamp) {
212  [installTracker clearPreviousInstallTimestamp];
213  }
214  } else if (previousTimestamp) {
215  SPSelfDescribingJson * installEvent = [[SPSelfDescribingJson alloc] initWithSchema:kSPApplicationInstallSchema andData:@{}];
216  SPUnstructured * event = [SPUnstructured build:^(id<SPUnstructuredBuilder> builder) {
217  [builder setEventData:installEvent];
218  [builder setTimestamp:previousTimestamp];
219  }];
220  [self track:event];
221  [installTracker clearPreviousInstallTimestamp];
222  }
223  }

521 - (void)track:(SPEvent *)event {
522  if (!event || !_dataCollection) return;
523 
524  if ([event isKindOfClass:SPScreenView.class]) {
525  @synchronized (_currentScreenState) {
526  SPScreenView *screenView = (SPScreenView *)event;
527  _previousScreenState = _currentScreenState;
528  _currentScreenState = [screenView getScreenState];
529  [screenView updateWithPreviousState: _previousScreenState];
530  }
531  }
532 
533  [event beginProcessingWithTracker:self];
534  [self processEvent:event];
535  [event endProcessingWithTracker:self];
536 }

546 - (SPPayload *)getFinalPayloadWithPayload:(SPPayload *)pb andContext:(NSMutableArray *)contextArray andEventId:(NSString *)eventId {
547  [pb addDictionaryToPayload:_trackerData];
548 
549  // Add Subject information
550  if (_subject != nil) {
551  [pb addDictionaryToPayload:[[_subject getStandardDict] getAsDictionary]];
552  }
553  [pb addValueToPayload:SPDevicePlatformToString(_devicePlatform) forKey:kSPPlatform];
554 
555  // Add the contexts
556  NSMutableArray<SPSelfDescribingJson *> *contexts = contextArray;
557  [self addBasicContextsToContexts:contexts eventId:eventId isService:NO]; // isService = NO is just the default - this method will be removed in the version 2.0
558 
559  [self wrapContexts:contexts toPayload:pb];
560  return pb;
561 }

563 - (SPPayload *)payloadWithEvent:(SPTrackerEvent *)event {
564  SPPayload *payload = [SPPayload new];
565  payload.allowDiagnostic = !event.isService;
566 
567  [self addBasicPropertiesToPayload:payload event:event];
568  if (event.isPrimitive) {
569  [self addPrimitivePropertiesToPayload:payload event:event];
570  } else {
571  [self addSelfDescribingPropertiesToPayload:payload event:event];
572  }
573  NSMutableArray<SPSelfDescribingJson *> *contexts = event.contexts;
574  [self addBasicContextsToContexts:contexts event:event];
575  [self addGlobalContextsToContexts:contexts event:event];
576  [self wrapContexts:contexts toPayload:payload];
577 
578  return payload;
579 }

616 - (void)addBasicContextsToContexts:(NSMutableArray<SPSelfDescribingJson *> *)contexts eventId:(NSString *)eventId isService:(BOOL)isService {
617  if (_subject) {
618  NSDictionary * platformDict = [[_subject getPlatformDict] getAsDictionary];
619  if (platformDict != nil) {
620  [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:_platformContextSchema andData:platformDict]];
621  }
622  NSDictionary * geoLocationDict = [_subject getGeoLocationDict];
623  if (geoLocationDict != nil) {
624  [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:kSPGeoContextSchema andData:geoLocationDict]];
625  }
626  }
627 
628  if (_applicationContext) {
629  SPSelfDescribingJson * contextJson = [SPUtilities getApplicationContext];
630  if (contextJson != nil) {
631  [contexts addObject:contextJson];
632  }
633  }
634 
635  if (isService) {
636  return;
637  }
638 
639  // Add session
640  if (_session) {
641  NSDictionary *sessionDict = [_session getSessionDictWithEventId:eventId];
642  if (sessionDict) {
643  [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:kSPSessionContextSchema andData:sessionDict]];
644  } else {
645  SPLogTrack(nil, @"Unable to get session context for eventId: %@", eventId);
646  }
647  }
648 
649  // Add screen context
650  if (_screenContext && _currentScreenState) {
651  SPSelfDescribingJson * contextJson = [SPUtilities getScreenContextWithScreenState:_currentScreenState];
652  if (contextJson != nil) {
653  [contexts addObject:contextJson];
654  }
655  }
656 
657  // Add GDPR context
658  if (self.gdpr) {
659  SPSelfDescribingJson *gdprContext = self.gdpr.context;
660  if (gdprContext) {
661  [contexts addObject:gdprContext];
662  }
663  }
664 }

672 - (void)wrapContexts:(NSArray<SPSelfDescribingJson *> *)contexts toPayload:(SPPayload *)payload {
673  if (contexts.count == 0) {
674  return;
675  }
676  NSMutableArray<NSDictionary *> *data = [NSMutableArray new];
677  for (SPSelfDescribingJson *context in contexts) {
678  [data addObject:[context getAsDictionary]];
679  }
680

683  return;
684  }
685  [payload addDictionaryToPayload:[finalContext getAsDictionary]
686  base64Encoded:_base64Encoded
687  typeWhenEncoded:kSPContextEncoded
688  typeWhenNotEncoded:kSPContext];
