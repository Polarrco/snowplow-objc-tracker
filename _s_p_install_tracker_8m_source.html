29 - (id) init {
30  if (self = [super init]) {
31  NSUserDefaults * userDefaults = [NSUserDefaults standardUserDefaults];
32  if ([userDefaults objectForKey:kSPInstalledBefore] == nil) {
33  // mark the install if there's no value in userDefaults
34  [userDefaults setObject:@YES forKey:kSPInstalledBefore];
35  [userDefaults setObject:[SPUtilities getTimestamp] forKey:kSPInstallTimestamp];
36  // since the value was missing in userDefaults, we're assuming this is a new install
37  self.isNewInstall = YES;
38  } else {
39  // if there's an object in standardUserDefaults - someone has been there!
40  self.isNewInstall = NO;
41  }
42  return self;
43  }
44  return nil;
45 }
