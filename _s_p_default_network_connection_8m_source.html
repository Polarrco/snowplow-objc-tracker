65 - (void) setup {
66  _dataOperationQueue.maxConcurrentOperationCount = _emitThreadPoolSize;
67  NSString *urlPrefix = _protocol == SPHttp ? @"http://" : @"https://";
68  NSString *urlSuffix = _httpMethod == SPRequestGet ? kSPEndpointGet : kSPEndpointPost;
69  if (_customPostPath && _httpMethod == SPRequestPost) {
70  urlSuffix = _customPostPath;
71  }
72  _urlEndpoint = [NSURL URLWithString:[NSString stringWithFormat:@"%@%@%@", urlPrefix, _urlString, urlSuffix]];
73 
74  if ([_urlEndpoint scheme] && [_urlEndpoint host]) {
75  SPLogDebug(@"Emitter URL created successfully '%@'", _urlEndpoint);
76  NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
77  [userDefaults setObject:_urlString forKey:kSPErrorTrackerUrl];
78  [userDefaults setObject:urlSuffix forKey:kSPErrorTrackerProtocol];
79  [userDefaults setObject:urlPrefix forKey:kSPErrorTrackerMethod];
80  } else {
81  SPLogDebug(@"Invalid emitter URL: '%@'", _urlEndpoint);
82  NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
83  [userDefaults setObject:@"acme.com" forKey:kSPErrorTrackerUrl];
84  [userDefaults setObject:kSPEndpointPost forKey:kSPErrorTrackerProtocol];
85  [userDefaults setObject:@"http://" forKey:kSPErrorTrackerMethod];

142 - (NSArray<SPRequestResult *> *)sendRequests:(NSArray<SPRequest *> *)requests {
143  NSMutableArray<SPRequestResult *> *results = [NSMutableArray new];
144 
145  for (SPRequest *request in requests) {
146  NSMutableURLRequest *urlRequest = _httpMethod == SPRequestGet
147  ? [self buildGetRequest:request]
148  : [self buildPostRequest:request];
149 
150  [_dataOperationQueue addOperationWithBlock:^{
151  //source: https://forums.developer.apple.com/thread/11519
152  __block NSHTTPURLResponse *httpResponse = nil;
153  __block NSError *connectionError = nil;
154  dispatch_semaphore_t sem;
155 
156  sem = dispatch_semaphore_create(0);
157 
158  [[[NSURLSession sharedSession] dataTaskWithRequest:urlRequest
159  completionHandler:^(NSData *data, NSURLResponse *urlResponse, NSError *error) {
160 
161  connectionError = error;
162  httpResponse = (NSHTTPURLResponse*)urlResponse;
163  dispatch_semaphore_signal(sem);
164  }] resume];
165 
166  dispatch_semaphore_wait(sem, DISPATCH_TIME_FOREVER);
167 
168  BOOL isSuccessful = [httpResponse statusCode] >= 200 && [httpResponse statusCode] < 300;
169  if (!isSuccessful) {
170  SPLogError(@"Connection error: %@", connectionError);
171  }
172  SPRequestResult *result = [[SPRequestResult alloc] initWithSuccess:isSuccessful storeIds:request.emitterEventIds];
173 
174  @synchronized (results) {
175  [results addObject:result];
176  }
177  }];
178  }
179  [_dataOperationQueue waitUntilAllOperationsAreFinished];
180  return results;
181 }

185 - (NSMutableURLRequest *)buildPostRequest:(SPRequest *)request {
186  NSData *requestData = [NSJSONSerialization dataWithJSONObject:[request.payload getAsDictionary] options:0 error:nil];
187  NSMutableURLRequest *urlRequest = [NSMutableURLRequest requestWithURL:[NSURL URLWithString:_urlEndpoint.absoluteString]];
188  [urlRequest setValue:[NSString stringWithFormat:@"%@", @(requestData.length).stringValue] forHTTPHeaderField:@"Content-Length"];
189  [urlRequest setValue:kSPAcceptContentHeader forHTTPHeaderField:@"Accept"];
190  [urlRequest setValue:kSPContentTypeHeader forHTTPHeaderField:@"Content-Type"];
191  [urlRequest setHTTPMethod:@"POST"];
192  [urlRequest setHTTPBody:requestData];
193  return urlRequest;
194 }
