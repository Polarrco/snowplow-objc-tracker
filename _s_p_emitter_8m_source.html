217 - (void) sendGuard {
218  if (_isSending) {
219  return;
220  }
221  @synchronized (self) {
222  if (!_isSending) {
223  _isSending = YES;
224  @try {
225  [self attemptEmit];
226  } @catch (NSException *exception) {

228  _isSending = NO;
229  }

234 - (void)attemptEmit {
235  if (!_eventStore.count) {
236  SPLogDebug(@"Database empty. Returning.", nil);
237  _isSending = NO;
238  return;
239  }
240 
241  NSArray<SPEmitterEvent *> *events = [_eventStore emittableEventsWithQueryLimit:_emitRange];
242  NSArray<SPRequest *> *requests = [self buildRequestsFromEvents:events];
243  NSArray<SPRequestResult *> *sendResults = [_networkConnection sendRequests:requests];
244 
245  SPLogVerbose(@"Processing emitter results.");
246 
247  NSInteger successCount = 0;
248  NSInteger failureCount = 0;
249  NSMutableArray<NSNumber *> *removableEvents = [NSMutableArray new];
250 
251  for (SPRequestResult *result in sendResults) {
252  NSArray<NSNumber *> *resultIndexArray = result.storeIds;
253  if (result.isSuccessful) {
254  successCount += resultIndexArray.count;
255  [removableEvents addObjectsFromArray:resultIndexArray];
256  } else {
257  failureCount += resultIndexArray.count;
258  }
259  }
260 
261  [_eventStore removeEventsWithIds:removableEvents];
262 
263  SPLogDebug(@"Success Count: %@", [@(successCount) stringValue]);
264  SPLogDebug(@"Failure Count: %@", [@(failureCount) stringValue]);
265 
266  if (_callback != nil) {
267  if (failureCount == 0) {
268  [_callback onSuccessWithCount:successCount];
269  } else {
270  [_callback onFailureWithCount:failureCount successCount:successCount];
271  }
272  }
273 
274  if (failureCount > 0 && successCount == 0) {
275  SPLogDebug(@"Ending emitter run as all requests failed.", nil);
276  [NSThread sleepForTimeInterval:5];
277  _isSending = NO;
278  return;
279  } else {
280  [self attemptEmit];
281  }
282 }

284 - (NSArray<SPRequest *> *)buildRequestsFromEvents:(NSArray<SPEmitterEvent *> *)events {
285  NSMutableArray<SPRequest *> *requests = [NSMutableArray new];
286  NSNumber *sendingTime = [SPUtilities getTimestamp];
287  SPRequestOptions httpMethod = _networkConnection.httpMethod;
288 
289  if (httpMethod == SPRequestGet) {
290  for (SPEmitterEvent *event in events) {
291  SPPayload *payload = event.payload;
292  [self addSendingTimeToPayload:payload timestamp:sendingTime];
293  BOOL oversize = [self isOversize:payload];
294  SPRequest *request = [[SPRequest alloc] initWithPayload:payload emitterEventId:event.storeId oversize:oversize];
295  [requests addObject:request];
296  }
297  } else {
298  for (int i = 0; i < events.count; i += _bufferOption) {
299  NSMutableArray<SPPayload *> *eventArray = [NSMutableArray new];
300  NSMutableArray<NSNumber *> *indexArray = [NSMutableArray new];
301 
302  for (int j = i; j < (i + _bufferOption) && j < events.count; j++) {
303  SPEmitterEvent *event = events[j];
304 
305  SPPayload *payload = event.payload;
306  NSNumber *emitterEventId = @(event.storeId);
307  [self addSendingTimeToPayload:payload timestamp:sendingTime];
308 
309  if ([self isOversize:payload]) {
310  SPRequest *request = [[SPRequest alloc] initWithPayload:payload emitterEventId:emitterEventId.longLongValue oversize:YES];
311  [requests addObject:request];
312 
313  } else if ([self isOversize:payload previousPayloads:eventArray]) {
314  SPRequest *request = [[SPRequest alloc] initWithPayloads:eventArray emitterEventIds:indexArray];
315  [requests addObject:request];
316 
317  // Clear collection and build a new POST
318  eventArray = [NSMutableArray new];
319  indexArray = [NSMutableArray new];
320 
321  // Build and store the request
322  [eventArray addObject:payload];
323  [indexArray addObject:emitterEventId];
324 
325  } else {
326  // Add event to collections
327  [eventArray addObject:payload];
328  [indexArray addObject:emitterEventId];
329  }
330  }
331 
332  // Check if all payloads have been processed
333  if (eventArray.count) {
334  SPRequest *request = [[SPRequest alloc] initWithPayloads:eventArray emitterEventIds:indexArray];
335  [requests addObject:request];
336  }
337  }
338  }
339  return requests;
340 }

351 - (BOOL)isOversize:(SPPayload *)payload byteLimit:(NSUInteger)byteLimit previousPayloads:(NSArray<SPPayload *> *)previousPayloads {
352  NSUInteger totalByteSize = payload.byteSize;
353  for (SPPayload *previousPayload in previousPayloads) {
354  totalByteSize += previousPayload.byteSize;
355  }
356  NSUInteger wrapperBytes = previousPayloads.count > 0 ? (previousPayloads.count + POST_WRAPPER_BYTES) : 0;
357  return totalByteSize + wrapperBytes > byteLimit;
358 }
